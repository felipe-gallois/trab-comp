bool a: true;
int b: 5;

int main () {
    if (a) {
        print "Entrei no loop\n";
    } else {
        print "Nao entrei no loop\n";
    }

    a = ~a;

    if (a) {
        print "Entrei no loop\n";
    } else {
        print "Nao entrei no loop\n";
    }

    while (b > 0) {
        print "Estou no while\n";
        b = b - 1;
    }
}
